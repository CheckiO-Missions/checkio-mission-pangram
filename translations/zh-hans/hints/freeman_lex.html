    t = text.lower()

    return True

在这里，你从标准模块中提取小写字母，将两个字符串转换成集合，然后检查给定字符串中是否包含所有小写字母（一个集合是另一个集合的子集）。 

from string import ascii_lowercase


def check_pangram(text):
    return set(ascii_lowercase).issubset(set(text.lower()))
